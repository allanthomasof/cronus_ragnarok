hugel,88,168,4	script	Eventos Incríveis::CoolEventCorpStaffHugel	874,{
	cutin "zonda_01",2;
	show "[Funcionário Corp. Eventos Incríveis]";
	show "Bem-vindo à Corp. Eventos Incríveis.";
	show "Nossos funcionários estão sempre trabalhando";
	show "para superar suas expectativas";
	show "de qualidade dos serviços. Então,";
	show "como posso ajudá-lo hoje?";
	next;
	switch(select("Salvar:Usar armazém:Alugar carrinho:Serviço de Senha para Armazém:Cancelar")){
	case 1:
		show "[Funcionário Corp. Eventos Incríveis]";
		show "Seu ponto de retorno";
		show "foi salvo aqui";
		show "nessa vila de Hugel.";
		show "Obrigado por usar a";
		show "Corp. Eventos Incríveis~";
		savepoint "hugel",96,155;
		close2;
		break;
	case 2:
		show "[Funcionário Corp. Eventos Incríveis]";
		if(basicskillcheck() && getskilllv("NV_BASIC") < 6){
			show "Desculpe, mas você";
			show "precisa das Habilidades Básicas";
			show "de Aprendiz Nível 6 para";
			show "usar o Serviço de Armazém.";
			close2;
			break;
		}
		if(Zeny < 40){
			show "Desculpe, mas você não";
			show "possui zeny suficiente para";
			show "o Serviço de Armazém. Nosso";
			show "acesso ao armazém custa 40 zeny.";
			close2;
			break;
		}
		set Zeny, Zeny - 40;
		show "Deixe-me abrir seu armazém";
		show "particular nesse exato momento.";
		show "Obrigado por apoiar a Corp.";
		show "Eventos Incríveis usando nossos";
		show "serviços. Tenha um bom dia~";
		callfunc("F_CheckKafCode");	//checa sua senha do armazém, se adicionado
		close2;
		openstorage;
		break;
	case 3:
		show "[Funcionário Corp. Eventos Incríveis]";
		if(BaseClass != Job_Merchant){
			show "Desculpe, mas o";
			show "serviço de aluguel de carrinho";
			show "é disponível apenas para Mercadores,";
			show "Ferreiros, Mestre-Ferreiros,";
			show "Alquimistas e Criadores.";
			close2;
			break;
		}
		if(getskilllv("MC_PUSHCART") == 0){
			show "Você só pode alugar carrinhos";
			show "após possuir a habilidade Usar Carrinho.";
			close2;
			break;
		}
		if(checkcart() == 1){
			show "Você já está";
			show "com o carrinho equipado.";
			show "Infelizmente, nós não";
			show "alugamos mais de um";
			show "carrinho por vez.";
			close2;
			break;
		}
		if(countitem(7061)){
			delitem 7061,1;
			setcart;
			close2;
			break;
		}
		show "O aluguel do carrinho";
		show "custa 800 zeny. Gostaria";
		show "de alugar um carrinho?";
		next;
		if(select("Alugar o carrinho:Cancelar") == 1){
			if(Zeny < 800){
				show "[Funcionário Corp. Eventos Incríveis]";
				show "Desculpe, mas você";
				show "não possui zeny suficiente";
				show "para pagar o aluguel";
				show "do carrinho que custa 800 zeny.";
				close2;
				break;
			} else {
				set Zeny, Zeny - 800;
				setcart;
				close2;
				break;
			}
		}
		close2;
		break;
	case 4:
		callfunc("F_SetKafCode","[Cool Event Corp. Staff]","Cool Event Corp.");
		//essa função fecha a mensagem e remove cutins
	case 5:
	default:
		show "[Funcionário Corp. Eventos Incríveis]";
		show "Corp. Eventos Incríveis está sempre";
		show "se esforçando para providenciar os melhores";
		show "serviços para nossos clientes.";
		show "Ajude-nos a ser os melhores";
		show "fornecendo-nos suas opiniões";
		show "e relato honesto. Obrigado.";
		close2;
	}
	cutin "", 255;
	end;
}
